//
//connect to device
//
device AMAP42KL-KBR
si SWD
speed 1000
r
sleep 10

//
// **** Change this based on device flashmap ****
// Load the image at a temp place in flash
//
// As of J-Link v7.70 (8/3/22), by default the loadbin command performs a
// device reset. Use noreset to avoid (ignored by older J-Link versions).
//
loadbin am_patch_ota.bin 0xFC000 noreset

//
// Set up OTA Descriptor at 0xFE000
//
w4 0xFE000 0xFC003
w4 0xFE004 0xFFFFFFFF

//
// Set up OTAPointer
//
w4 0x40020264 0xFE001

//
// Reset
//
w4 0x40000004 0x1B
sleep 2000
connect
mem32 0xFE000, 1
qc
