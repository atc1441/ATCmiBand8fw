248// OTA Protocol between OTA application and SecureBoot
249// OTAPOINTER will be initialized as follows:
250// Most significant 30 bits will correspond to most significant 30 bits of OTA Descriptor
251// Least Significant bit (bit 0) should be initialized to 1 to indicate a valid OTA Descriptor
252// bit 1 should be initialized to 1 to indicate that the list contains an SBL OTA
253// OTA Descriptor points to a list of entries, each corresponding to an OTA blob, list terminating in 0xFFFFFFFF
254// Each list entry word comprises of following:
255// Most significant 30 bits will correspond to most significant 30 bits of OTA blob pointer
256// Least Significant 2 bits should be initialized to 1 to indicate a valid OTA Pending
257// After Secboot processes an OTA, it clears the least significant bit (bit 0)
258// bit 1 indicates the status of the OTA - 0 for Success, 1 for Failure
259
260// Store the b1 to store the return status - pass/fail
261// b0 is used to indicate pending/done
262#define AM_HAL_OTA_VALID_MASK 0x3
263// Use bit 27, 29, 30, 31 to store the return status, if failure
264#define AM_HAL_OTA_STATUS_MASK 0xE8000000
265#define AM_HAL_OTA_GET_BLOB_PTR(ptr) (((uint32_t)(ptr) & ~(AM_HAL_OTA_VALID_MASK | AM_HAL_OTA_STATUS_MASK)))
266#define AM_HAL_OTA_IS_VALID(ptr) (((uint32_t)(ptr) & AM_HAL_OTA_VALID_MASK) == AM_HAL_OTA_VALID_MASK)
267#define AM_HAL_OTA_IS_IN_PROGRESS(ptr) (((uint32_t)(ptr) & AM_HAL_OTA_VALID_MASK) == AM_HAL_OTA_DONE_STATUS_IN_PROGRESS)
268
269#define AM_HAL_OTA_DONE_STATUS_MASK 0x3
270#define AM_HAL_OTA_DONE_STATUS_IN_PROGRESS 0x1
271#define AM_HAL_OTA_DONE_STATUS_FAILURE 0x2
272#define AM_HAL_OTA_DONE_STATUS_SUCCESS 0x0
