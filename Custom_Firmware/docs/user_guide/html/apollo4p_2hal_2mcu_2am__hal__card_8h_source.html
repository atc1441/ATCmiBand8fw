376static inline uint32_t am_hal_unstuff_bits(uint32_t *resp, uint32_t start, uint32_t size)
377{
378 const int __size = size;
379 const uint32_t __mask = (__size < 32 ? 1 << __size : 0) - 1;
380 const int __off = 3 - ((start) / 32);
381 const int __shft = (start)&31;
382 uint32_t __res;
383
384 __res = resp[__off] >> __shft;
385 if (__size + __shft > 32)
386 {
387 __res |= resp[__off - 1] << ((32 - __shft) % 32);
388 }
389 return __res &__mask;
390}
391
392static inline uint32_t am_hal_unstuff_bytes(uint32_t *ext_csd, uint32_t start, uint32_t size)
393{
394 int i;
395 const uint32_t __size = size;
396 uint32_t __res = 0x0;
397 const uint8_t *__p = (const uint8_t *)ext_csd;
398 for (i = 0; i < __size; i++)
399 {
400 __res |= __p[start + i] << 8 * i;
401 }
402 return __res;
403}
