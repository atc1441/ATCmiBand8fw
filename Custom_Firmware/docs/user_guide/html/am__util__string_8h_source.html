67#define AM_CATTR_NONE 0x00
68#define AM_CATTR_ALPHA 0x01
69#define AM_CATTR_LOWER 0x02
70#define AM_CATTR_UPPER 0x04
71#define AM_CATTR_DIGIT 0x08
72#define AM_CATTR_XDIGIT 0x10
73#define AM_CATTR_WHSPACE 0x20
74#define AM_CATTR_FILENM83 0x80
75
76//
77// Set MINIMIZE_CATTR_TABLE to 1 to configure for minimal CATTR table size,
78// (256 instead of 512 bytes) but at a cost of slightly larger code size.
79// However, setting this option also provides an additional level of checking
80// of the argument; if the argument is not a uint8_t, the functions are
81// guaranteed to return 0.
82//
83#define MINIMIZE_CATTR_TABLE 0
