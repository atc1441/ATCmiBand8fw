216// OTA Protocol between OTA application and SecureBoot
217// OTAPOINTER will be initialized as follows:
218// Most significant 30 bits will correspond to most significant 30 bits of OTA Descriptor
219// Least Significant bit (bit 0) should be initialized to 1 to indicate a valid OTA Descriptor
220// bit 1 should be initialized to 1 to indicate that the list contains an SBL OTA
221// OTA Descriptor points to a list of entries, each corresponding to an OTA blob, list terminating in 0xFFFFFFFF
222// Each list entry word comprises of following:
223// Most significant 30 bits will correspond to most significant 30 bits of OTA blob pointer
224// Least Significant 2 bits should be initialized to 1 to indicate a valid OTA Pending
225// After Secboot processes an OTA, it clears the least significant bit (bit 0)
226// bit 1 indicates the status of the OTA - 0 for Success, 1 for Failure
227
228// Store the b1 to store the return status - pass/fail
229// b0 is used to indicate pending/done
230#define AM_HAL_OTA_VALID_MASK 0x3
231// Use bit 27, 29, 30, 31 to store the return status, if failure
232#define AM_HAL_OTA_STATUS_MASK 0xE8000000
233#define AM_HAL_OTA_GET_BLOB_PTR(ptr) (((uint32_t)(ptr) & ~(AM_HAL_OTA_VALID_MASK | AM_HAL_OTA_STATUS_MASK)))
234#define AM_HAL_OTA_IS_VALID(ptr) (((uint32_t)(ptr) & AM_HAL_OTA_VALID_MASK) == AM_HAL_OTA_VALID_MASK)
235#define AM_HAL_OTA_IS_IN_PROGRESS(ptr) (((uint32_t)(ptr) & AM_HAL_OTA_VALID_MASK) == AM_HAL_OTA_DONE_STATUS_IN_PROGRESS)
236
237#define AM_HAL_OTA_DONE_STATUS_MASK 0x3
238#define AM_HAL_OTA_DONE_STATUS_IN_PROGRESS 0x1
239#define AM_HAL_OTA_DONE_STATUS_FAILURE 0x2
240#define AM_HAL_OTA_DONE_STATUS_SUCCESS 0x0
