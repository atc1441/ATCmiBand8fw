57// For direct 32-bit access to a register or memory location, use AM_REGVAL:
58// AM_REGVAL(0x1234567) |= 0xDEADBEEF;

60//*****************************************************************************

62#define AM_REGVAL_FLOAT(x) (*((volatile float *)(x)))

64//*****************************************************************************
65//
66// AM_REGADDR()
67// One thing CMSIS does not do well natively is to provide for static register
68// address computation. The address-of operator (e.g. &periph->reg) is helpful,
69// but does run into problems, such as when attempting to cast the resulting
70// pointer to a uint32_t. The standard C macro, offsetof() can help.
71//
72// Use AM_REGADDR() for single-module peripherals.
73// Use AM_REGADDRn() for multi-module peripherals (e.g. IOM, UART).
74//
75//*****************************************************************************
76#define AM_REGADDR(periph, reg) ( periph##_BASE + offsetof(periph##_Type, reg) )
77
78#define AM_REGADDRn(periph, n, reg) ( periph##0_BASE + \
79 offsetof(periph##0_Type, reg) + \
80 (n * (periph##1_BASE - periph##0_BASE)) )
81

83//
84// Bitfield macros.
85//
86//*****************************************************************************
87#ifdef CMSIS_BFNOT
88#define AM_BFW(module, reg, field, value) \
89 (module->reg = ((module->reg & ~(module##_##reg##_##field##_Msk)) | \
90 _VAL2FLD(module##_##reg##_##field, value)))
91
92#define AM_BFR(module, reg, field) \
93 _FLD2VAL(module##_##reg##_##field, module->reg)
94
95#define AM_BFRn(module, instance, reg, field) \
96 (_FLD2VAL(module##0_##reg##_##field, \
97 *((volatile uint32_t *)(AM_REGADDRn(module, instance, reg)))) )
98
99#define AM_BFWn(module, instance, reg, field, value) \
100 (*((volatile uint32_t *)(AM_REGADDRn(module, instance, reg))) = \
101 (((*((volatile uint32_t *)(AM_REGADDRn(module, instance, reg)))) & \
102 ~(module##0_##reg##_##field##_Msk) ) | \
103 _VAL2FLD(module##0_##reg##_##field, value)) )
104
105#endif

107//*****************************************************************************
108//
109// Critical section assembly macros
110//
111// These macros implement critical section protection using inline assembly
112// for various compilers. They are intended to be used in other register
113// macros or directly in sections of code.
114//
115// Important usage note: These macros create a local scope and therefore MUST
116// be used in pairs.
117//
118//*****************************************************************************
119#define AM_CRITICAL_BEGIN \
120 if ( 1 ) \
121 { \
122 volatile uint32_t ui32Primask_04172010; \
123 ui32Primask_04172010 = am_hal_interrupt_master_disable();
124
125#define AM_CRITICAL_END \
126 am_hal_interrupt_master_set(ui32Primask_04172010); \
127 }
