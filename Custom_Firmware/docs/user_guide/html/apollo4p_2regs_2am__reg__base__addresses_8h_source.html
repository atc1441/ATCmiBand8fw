94// ****************************************************************************
95// RAM Memory Map:
96// TCM: 384KB 0x10000000 - 0x1005FFFF
97// Shared System SRAM (SSRAM): 1024KB 0x10060000 - 0x1015FFFF
98// Extended RAM: 384KB 0x10160000 - 0x101BFFFF
99// Shared System SRAM (SSRAM): 1024KB 0x101C0000 - 0x102BFFFF
100// ****************************************************************************
101//
102// TCM address space
103//
104#define SRAM_BASEADDR (0x10000000UL)
105#define TCM_BASEADDR SRAM_BASEADDR
106#define TCM_MAX_SIZE (384UL * 1024UL)
107
108//
109// The two non-contiguous SSRAM spaces are referred to as SSRAM0 and SSRAM1.
110// SSRAM0 address space.
111//
112#define SSRAM0_BASEADDR (SRAM_BASEADDR + TCM_MAX_SIZE)
113#define SSRAM0_MAX_SIZE (1024UL * 1024UL)
114
115//
116// Extended RAM address space.
117// DSP0 layout is exactly as Apollo4b. DSP1RAMs were replaced in Apollo4p with SSRAM1.
118//
119#define EXTRAM_BASEADDR (SSRAM0_BASEADDR + SSRAM0_MAX_SIZE)
120#define DSP0IRAM_BASEADDR EXTRAM_BASEADDR
121#define DSP0IRAM_MAX_SIZE (128UL * 1024UL)
122#define DSP0DRAM_BASEADDR (DSP0IRAM_BASEADDR + DSP0IRAM_MAX_SIZE)
123#define DSP0DRAM_MAX_SIZE (256UL * 1024UL)
124#define EXTRAM_MAX_SIZE (DSP0IRAM_MAX_SIZE + DSP0DRAM_MAX_SIZE)
125
126//
127// SSRAM1 address space
128//
129#define SSRAM1_BASEADDR (EXTRAM_BASEADDR + EXTRAM_MAX_SIZE)
130#define SSRAM1_MAX_SIZE (1024UL * 1024UL)
131
132//
133// RAM sizes.
134// NONTCM_MAX_SIZE is the total of contiguous RAM after TCM (mix of SSRAM and Extended)
135// SSRAM_MAX_SIZE is the total of Shared System RAM, but is not necessarily contiguous.
136// RAM_TOTAL_SIZE is the grand total of all contiguous RAM.
137//
138#define SSRAM_MAX_SIZE (SSRAM0_MAX_SIZE + SSRAM1_MAX_SIZE)
139#define NONTCM_MAX_SIZE (EXTRAM_MAX_SIZE + SSRAM_MAX_SIZE)
140#define RAM_TOTAL_SIZE (TCM_MAX_SIZE + NONTCM_MAX_SIZE)
141

146#define MRAM_MAX_SIZE (2048UL * 1024UL)
